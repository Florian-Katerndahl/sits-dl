@torch.compile(fullgraph=True, dynamic=False)
def foo(a: torch.tensor):
    b = torch.reshape(a, (-1, a.shape[2], a.shape[3]))
    normed = (b - torch.mean(b, dim=2, keepdim=True)) / torch.std(b, dim=2, keepdim=True) + 1e-4
    return normed
